@startuml

title Conditional - Score into Grade 


start

:get user input score; 

if( 90<score<=100 )
    :Grade "A";
    
elseif( 80<score<=90 )
    :Grade "B";
    
elseif( 70<score<=80 )
    :Grade "C";
    
elseif( 60<score<=70 )
    :Grade "D";
    
elseif( 0<=score<=60 )
    :Grade "F";
    
    
endif
:print grade;

stop

@enduml
